$("#dtBox").DateTimePicker({
    maxDate: '0',
    dateFormat: "dd-MMM-yyyy",

    addEventHandlers: function()
    {
        var oDTP = this;

        oDTP.settings.minDate = oDTP.getDateTimeStringInFormat("Date", "dd-MMM-yyyy", new Date());
    }
});
